## ⚡ Quick-Start Scripts

<Accordion title="Shell Quick Start">
  <CodeBlock language="bash">
{`# Install necessary tools and dependencies
yarn init -y
nvm use 18.18.0
yarn install

# Download and set up the TEE simulator
curl -L -O https://github.com/Leecheal/tappd-simulator/releases/download/v0.1.4/tappd-simulator-0.1.4-aarch64-apple-darwin.tgz
tar -xvf tappd-simulator-0.1.4-aarch64-apple-darwin.tgz
chmod +x run_simulator.sh
./run_simulator.sh &

# Authenticate with Phala Cloud (replace <TOKEN> with your Phala token)
phala auth login <TOKEN>

# Create Dockerfile and docker-compose.yml files (or use your own)

# Build and push Docker image
docker build -t <IMAGE_NAME> .
docker push <IMAGE_NAME>:latest

# Multi-platform image build (optional)
docker buildx build --platform linux/amd64,linux/arm64 -t <IMAGE_NAME>:latest --push --provenance=false .

# Update docker-compose if needed
chmod +x update_compose.sh
./update_compose.sh

# Deploy to Phala Cloud
phala cvms create --name <your-cvm-name> --compose docker-compose.yml --skip-env
`}
  </CodeBlock>
</Accordion>

<Accordion title="Command Script: update_compose.sh">
  <CodeBlock language="bash">
{`#!/bin/bash

# update_compose.sh - Create or update docker-compose.yml with port mapping

set -e

echo "========================================================"
echo "Updating Docker Compose Configuration"
echo "========================================================"

# Backup existing docker-compose.yml if it exists
if [ -f "docker-compose.yml" ]; then
  echo "Backing up existing docker-compose.yml to docker-compose.yml.bak"
  cp docker-compose.yml docker-compose.yml.bak
fi

# Generate a new docker-compose.yml using Phala CLI
phala docker generate -o docker-compose.yml

# Verify successful generation
if [ $? -ne 0 ]; then
  echo "Error: Failed to generate docker-compose.yml"
  exit 1
fi

# Add generic port mapping and remove PORT env
TMP_FILE=$(mktemp)
cat docker-compose.yml | awk '
  {
    if ($0 ~ /container_name: app/) {
      print $0;
      print "    ports:";
      print "      - \"8080:8080\"";
      next;
    } else if ($0 ~ /- PORT=\${PORT}/) {
      next;  # Remove PORT env variable
    } else {
      print $0;
    }
  }
' > "$TMP_FILE"
mv "$TMP_FILE" docker-compose.yml

echo "Docker Compose file updated successfully."
echo "You can now deploy your CVM with 'phala cvms create'."
echo "========================================================"

cat docker-compose.yml
`}
  </CodeBlock>
</Accordion>
